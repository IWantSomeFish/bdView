## Задача 2. API для управления моделью

Добавь три эндпоинта в `api.controller.ts` и зарегистрируй маршруты.

### 2.1 Контроллер — добавь три метода в `ApiController`

```typescript
import { loadModel, WifiModel } from '../utils/model/model.inference';
import { trainWifiModel, serializeModel, ScanRow } from '../utils/model/model.train';
import * as fs from 'fs';
import * as path from 'path';

// Путь к файлу модели (относительно запуска сервера)
const MODEL_PATH = path.resolve('./wifi-model.json');

// GET /models — список загруженных моделей
async listModels(_req: Request, res: Response) {
    const raw = JSON.parse(fs.readFileSync(MODEL_PATH, 'utf-8'));
    res.json([{
        id:          raw.version,
        name:        `${raw.type}.json`,
        description: raw.type,
        version:     raw.version,
        uploadedAt:  raw.createdAt,
        status:      'ok' as const,
    }]);
}

// POST /models/upload — загрузить JSON-файл модели (multipart, поле "model")
async uploadModel(req: Request, res: Response) {
    if (!req.file) {
        return res.status(400).json({ error: 'Нужен файл модели (поле "model")' });
    }
    const json = req.file.buffer.toString('utf-8');
    const parsed = JSON.parse(json);        // выбросит ошибку если невалидный JSON
    loadModel(parsed);                      // проверит обязательные поля
    fs.writeFileSync(MODEL_PATH, json);
    res.json({ ok: true, version: parsed.version });
}

// POST /models/train — обучить новую модель на загруженной БД
async trainModel(req: Request, res: Response) {
    if (!req.file) {
        return res.status(400).json({ error: 'Нужна БД (поле "database")' });
    }
    // достать сырые сканы из БД
    const rawDB = await service.getRawScans(req.file.buffer);   // см. 2.2
    const model = trainWifiModel(rawDB, {
        minAppearanceFreq:     Number(req.body.minAppearanceFreq)     || 0.5,
        maxVariance:           Number(req.body.maxVariance)           || 120,
        minSpatialConsistency: Number(req.body.minSpatialConsistency) || 0.35,
    });
    fs.writeFileSync(MODEL_PATH, serializeModel(model));
    res.json({ ok: true, version: model.version, metrics: model.metrics });
}
```

### 2.2 Добавить `getRawScans` в `mainService` (`api.service.ts`)

```typescript
async getRawScans(buffer: Buffer): Promise<ScanRow[]> {
    const rawDB = await this.repo.dump(buffer);
    const networks = (rawDB['wifi_networks'] ?? []) as any[];
    return networks.map(n => ({
        bssid:     String(n.bssid ?? ''),
        signal:    Number(n.signal ?? -100),
        latitude:  Number(n.latitude ?? 0),
        longitude: Number(n.longitude ?? 0),
        runId:     String(n.runId ?? n.calibrationRunId ?? ''),
    })).filter(n => n.bssid && n.runId);
}
```

> Названия колонок (`bssid`, `signal`, `latitude`, `longitude`, `runId`) уточни
> через `PRAGMA table_info(wifi_networks)` — они могут отличаться в конкретной БД.

## Задача 3. Фронтенд — убрать заглушки в `ModelsTab.tsx`

Замени весь файл `frontend/src/presentation/components/ModelsTab.tsx`:

```typescript
import React, { useState, useEffect, useRef } from 'react';

interface ModelEntry {
  id: string; name: string; description: string;
  version: string; uploadedAt: string; status: 'ok' | 'failed';
}

const API = 'http://localhost:3000';

const ModelsTab: React.FC = () => {
  const [models, setModels]     = useState<ModelEntry[]>([]);
  const [loading, setLoading]   = useState(false);
  const [message, setMessage]   = useState<string | null>(null);
  const fileRef = useRef<HTMLInputElement>(null);

  const fetchModels = () => {
    fetch(`${API}/models`).then(r => r.json()).then(setModels).catch(() => {});
  };

  useEffect(() => { fetchModels(); }, []);

  const handleUpload = async () => {
    const file = fileRef.current?.files?.[0];
    if (!file) return;
    setLoading(true); setMessage(null);
    const form = new FormData();
    form.append('model', file);
    const r = await fetch(`${API}/models/upload`, { method: 'POST', body: form });
    const data = await r.json();
    setMessage(r.ok ? `Загружена версия ${data.version}` : data.error);
    if (r.ok) fetchModels();
    setLoading(false);
  };

  const handleTrain = async (dbFile: File) => {
    setLoading(true); setMessage(null);
    const form = new FormData();
    form.append('database', dbFile);
    const r = await fetch(`${API}/models/train`, { method: 'POST', body: form });
    const data = await r.json();
    setMessage(r.ok ? `Обучена версия ${data.version}, F1=${data.metrics?.f1}` : data.error);
    if (r.ok) fetchModels();
    setLoading(false);
  };

  const btnStyle = (dis: boolean): React.CSSProperties => ({
    padding: '10px 20px', backgroundColor: dis ? '#ccc' : '#007bff',
    color: 'white', border: 'none', borderRadius: '4px',
    cursor: dis ? 'not-allowed' : 'pointer', whiteSpace: 'nowrap',
  });

  return (
    <>
      <h2 style={{ marginTop: 0 }}>Модели нейросети</h2>

      {message && (
        <div style={{ padding: '10px', marginBottom: '16px', background: '#e8f4e8', borderRadius: '6px' }}>
          {message}
        </div>
      )}

      <div style={{ border: '1px solid var(--border)', borderRadius: '6px', padding: '16px', marginBottom: '16px' }}>
        <h3 style={{ marginTop: 0 }}>Загрузить модель (JSON)</h3>
        <div style={{ display: 'flex', gap: '10px', alignItems: 'center' }}>
          <input type="file" accept=".json" ref={fileRef} />
          <button disabled={loading} onClick={handleUpload} style={btnStyle(loading)}>
            Загрузить
          </button>
        </div>
      </div>

      <div style={{ border: '1px solid var(--border)', borderRadius: '6px', padding: '16px', marginBottom: '24px' }}>
        <h3 style={{ marginTop: 0 }}>Обучить модель на своей БД</h3>
        <div style={{ display: 'flex', gap: '10px', alignItems: 'center' }}>
          <input type="file" accept=".db,.sqlite"
            onChange={e => { if (e.target.files?.[0]) handleTrain(e.target.files[0]); }} />
          <span style={{ color: '#666', fontSize: 13 }}>
            Выбери .db файл — модель обучится и сохранится автоматически
          </span>
        </div>
      </div>

      <h3>Список моделей</h3>
      <table style={{ width: '100%', borderCollapse: 'collapse', fontSize: '14px' }}>
        <thead>
          <tr>
            {['Название', 'Описание', 'Версия', 'Дата', 'Статус'].map(col => (
              <th key={col} style={{ border: '1px solid var(--border)', padding: '8px', background: 'var(--bg)', textAlign: 'left' }}>{col}</th>
            ))}
          </tr>
        </thead>
        <tbody>
          {models.length === 0 && (
            <tr><td colSpan={5} style={{ padding: '12px', color: '#999', textAlign: 'center' }}>
              Нет загруженных моделей
            </td></tr>
          )}
          {models.map(m => (
            <tr key={m.id}>
              <td style={{ border: '1px solid var(--border)', padding: '8px' }}>{m.name}</td>
              <td style={{ border: '1px solid var(--border)', padding: '8px' }}>{m.description}</td>
              <td style={{ border: '1px solid var(--border)', padding: '8px', fontFamily: 'monospace' }}>{m.version}</td>
              <td style={{ border: '1px solid var(--border)', padding: '8px' }}>{m.uploadedAt}</td>
              <td style={{ border: '1px solid var(--border)', padding: '8px' }}>
                <span style={{ color: m.status === 'ok' ? '#27ae60' : '#e74c3c', fontWeight: 'bold' }}>
                  {m.status === 'ok' ? '✓ Активна' : '✗ Ошибка'}
                </span>
              </td>
            </tr>
          ))}
        </tbody>
      </table>
    </>
  );
};

export default ModelsTab;
```
